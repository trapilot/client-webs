# ===== 2. PRODUCT MODELS =====

class Product(models.Model):
    """Thang máy - Sản phẩm chính"""
    STATUS_CHOICES = [
        ('draft', 'Nháp'),
        ('published', 'Xuất Bản'),
        ('archived', 'Lưu Trữ'),
    ]

    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    category = models.ForeignKey(ElevatorCategory, on_delete=models.CASCADE, related_name='products', verbose_name='Danh Mục')
    
    # Thông tin cơ bản
    name = models.CharField('Tên Sản Phẩm', max_length=200)
    slug = models.SlugField('Slug', unique=True)
    product_code = models.CharField('Mã Sản Phẩm', max_length=100, unique=True)
    
    # Mô tả
    short_description = models.CharField('Mô Tả Ngắn', max_length=500)
    description = RichTextField('Mô Tả Chi Tiết')
    
    # Hình ảnh
    featured_image = models.ImageField('Hình Ảnh Chính', upload_to='products/')
    gallery = models.ManyToManyField('ProductImage', blank=True, related_name='products', verbose_name='Thư Viện Ảnh')
    
    # Thông số kỹ thuật
    capacity = models.IntegerField('Công Suất (kg)', validators=[MinValueValidator(100), MaxValueValidator(10000)])
    speed = models.DecimalField('Tốc Độ (m/s)', max_digits=3, decimal_places=1, validators=[MinValueValidator(0.1)])
    max_height = models.IntegerField('Chiều Cao Tối Đa (m)', validators=[MinValueValidator(3), MaxValueValidator(100)])
    power_consumption = models.CharField('Tiêu Thụ Điện', max_length=100, default='100 kWh/tháng')
    energy_saving = models.IntegerField('Tiết Kiệm Điện (%)', default=30, validators=[MinValueValidator(0), MaxValueValidator(100)])
    
    # Giá
    base_price = models.DecimalField('Giá Cơ Bản (đ)', max_digits=15, decimal_places=0, validators=[MinValueValidator(100000)])
    discount_price = models.DecimalField('Giá Giảm (đ)', max_digits=15, decimal_places=0, null=True, blank=True)
    
    # Badge
    is_popular = models.BooleanField('Phổ Biến', default=False)
    is_new = models.BooleanField('Mới', default=False)
    is_bestseller = models.BooleanField('Bestseller', default=False)
    
    # Warranty
    warranty_years = models.IntegerField('Bảo Hành (năm)', default=5)
    
    # Status
    status = models.CharField('Trạng Thái', max_length=20, choices=STATUS_CHOICES, default='published')
    is_featured = models.BooleanField(_(u'Featured'), default=False)
    
    # Meta
    seo_title = models.CharField('Tiêu Đề SEO', max_length=200, blank=True)
    seo_description = models.CharField('Mô Tả SEO', max_length=500, blank=True)
    seo_keywords = models.CharField('Từ Khóa SEO', max_length=300, blank=True)
    
    created_at = models.DateTimeField('Ngày Tạo', auto_now_add=True)
    updated_at = models.DateTimeField('Cập Nhật', auto_now=True)
    created_by = models.ForeignKey(User, on_delete=models.SET_NULL, null=True, related_name='created_products')

    class Meta:
        verbose_name = 'Product'
        verbose_name_plural = 'Products'
        ordering = ['-is_featured', '-created_at']
        indexes = [
            models.Index(fields=['slug']),
            models.Index(fields=['product_code']),
            models.Index(fields=['category']),
        ]

    def __str__(self):
        return self.name

    @property
    def get_discount_percent(self):
        if self.discount_price:
            discount = ((self.base_price - self.discount_price) / self.base_price) * 100
            return int(discount)
        return 0
